## <a name="example"></a><span data-ttu-id="49b0f-107">Příklad</span><span class="sxs-lookup"><span data-stu-id="49b0f-107">Example</span></span>

class A  
{  
   public virtual void Finalize() {}   // CS0465

{  
   ~B() {}  
}
