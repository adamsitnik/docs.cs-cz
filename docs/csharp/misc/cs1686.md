 <span data-ttu-id="fcc50-106">Následující ukázka generuje CS1686.</span><span class="sxs-lookup"><span data-stu-id="fcc50-106">The following sample generates CS1686.</span></span>

// CS1686.cs

class MyClass  
{  
   public unsafe delegate int * MyDelegate();  
  
   public unsafe int * Test()  
   {  
      int j = 0;  
      MyDelegate d = delegate { return &j; };   // CS1686  
      return &j;   // OK  
   }  
}
