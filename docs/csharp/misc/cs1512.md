 <span data-ttu-id="6664d-104">[Základní](../../csharp/language-reference/keywords/base.md) – klíčové slovo byl použit mimo metoda, vlastnost nebo konstruktoru.</span><span class="sxs-lookup"><span data-stu-id="6664d-104">The [base](../../csharp/language-reference/keywords/base.md) keyword was used outside of a method, property, or constructor.</span></span>

```csharp  
// CS1512.cs  
using System;  
  
class Base {}  
  
class CMyClass : Base  
{  
    private String xx = base.ToString();   // CS1512  
    // Try putting this initialization in the constructor instead:  
    // public CMyClass()  
    // {  
    //    xx = base.ToString();  
    // }  
  
    public static void Main()  
    {  
        CMyClass z = new CMyClass();  
    }  
}  
```
