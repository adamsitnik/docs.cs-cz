 <span data-ttu-id="1804b-106">Následující ukázka generuje CS0214:</span><span class="sxs-lookup"><span data-stu-id="1804b-106">The following sample generates CS0214:</span></span>  
  
```csharp  
// CS0214.cs  
// compile with: /target:library /unsafe  
public struct S  
{  
   public int a;  
}  
  
public class MyClass  
{  
   public static void Test()  
   {  
      S s = new S();  
      S * s2 = &s;    // CS0214  
      s2->a = 3;      // CS0214  
      s.a = 0;  
   }  
  
   // OK  
   unsafe public static void Test2()  
   {  
      S s = new S();  
      S * s2 = &s;  
      s2->a = 3;  
      s.a = 0;  
   }  
}  
```
