 <span data-ttu-id="e0882-107">Následující ukázka generuje CS0031 v kontextech zaškrtnuto a nezaškrtnuto:</span><span class="sxs-lookup"><span data-stu-id="e0882-107">The following sample generates CS0031 in both checked and unchecked contexts:</span></span>

{  
   public class a  
   {  
      public static void Main()  
      {  
         int num = (int)2147483648M; //CS0031  
         // Try using a larger numeric type instead:  
         // long num = (long)2147483648M; //CS0031

            unchecked  
            {  
                const byte b = (byte)d; // CS0031  
                // For small values try using a signed byte instead:  
                // const sbyte b = (sbyte)d;  
            }  
      }  
   }  
}
