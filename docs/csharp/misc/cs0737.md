## <a name="example"></a><span data-ttu-id="edca8-109">Příklad</span><span class="sxs-lookup"><span data-stu-id="edca8-109">Example</span></span>

```csharp  
// cs0737.cs

{  
    // Default access of private with no modifier.  
    int Return42();  
    // Try the following line instead.  
    // public int Return42();  
}

{  
    int Return42() { return (42); }  
}

{  
    public static int Main(string[] args)  
    {  
        Struct1 s1 = new Struct1();  
  
        return (1);  
    }  
  
}  
```
