## <a name="example"></a><span data-ttu-id="60582-107">Příklad</span><span class="sxs-lookup"><span data-stu-id="60582-107">Example</span></span>  
 <span data-ttu-id="60582-108">Následující příklad generuje CS0838:</span><span class="sxs-lookup"><span data-stu-id="60582-108">The following example generates CS0838:</span></span>  
  
```csharp  
// cs0838.cs  
using System;  
using System.Linq;  
using System.Linq.Expressions;  
  
namespace TestNamespace  
{  
    class Test  
    {  
        static int Main()  
        {  
  
            Expression<Func<int[,]>> expr =  
                () => new int[2, 2] { { 1, 2 }, { 3, 4 } }; // CS0838  
  
            // try the following 2 lines instead  
            int[,] nums = new int[2, 2] { { 1, 2 }, { 3, 4 } };  
            Expression<Func<int[,]>> expr2 = () => nums;   
  
            return 1;  
        }  
    }  
}  
```
