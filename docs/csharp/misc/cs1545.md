## <a name="example"></a><span data-ttu-id="a3d80-106">Příklad</span><span class="sxs-lookup"><span data-stu-id="a3d80-106">Example</span></span>  
  
```cpp  
// CPP1545.cpp  
// compile with: /clr /LD  
// a Visual C++ program  
using namespace System;  
public ref struct Employee {  
   Employee( String^ s, int d ) {}  
  
   property String^ name {  
      String^ get() {  
         return nullptr;  
      }  
   }  
};  
  
public ref struct Manager {  
   property Employee^ Report [String^] {  
      Employee^ get(String^ s) {  
         return nullptr;  
      }  
  
      void set(String^ s, Employee^ e) {}  
   }  
};  
```
