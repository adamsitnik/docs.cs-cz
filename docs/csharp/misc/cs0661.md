 <span data-ttu-id="5aecc-106">Následující ukázka generuje CS0661:</span><span class="sxs-lookup"><span data-stu-id="5aecc-106">The following sample generates CS0661:</span></span>  
  
```csharp  
// CS0661.cs  
// compile with: /W:3  
class Test   // CS0661  
{  
   public static bool operator == (object o, Test t)  
   {  
      return true;  
   }  
  
   public static bool operator != (object o, Test t)  
   {  
      return true;  
   }  
  
   public override bool Equals(object o)  
   {  
      return true;  
   }  
  
   // uncomment the GetHashCode function to resolve  
   // public override int GetHashCode()  
   // {  
   //    return 0;  
   // }  
  
   public static void Main()  
   {  
   }  
}  
```
